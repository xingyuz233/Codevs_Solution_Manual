#include <iostream>
#include <set>
using namespace std;
set<int> s;
set<int>::iterator itl, itr;
int main() {
    int n;
    int a, b;
    int ans = 0;
    int flag = 0;
    cin >> n;
    for (int i = 0; i < n; i++) {
        cin >> a >> b;
        if (!s.empty() && flag != a) {
            itr = s.upper_bound(b);
            if (itr == s.begin()) {
                ans += *itr - b;
                s.erase(itr);
            } else {
                itl = itr;
                itl--;
                if (itr == s.end()) {
                    ans += b - *itl;
                    s.erase(itl);
                } else {
                    if (b - *itl <= *itr - b) {
                        ans += b - *itl;
                        s.erase(itl);
                    } else {
                        ans += *itr - b;
                        s.erase(itr);
                    }
                }
            }
            ans %= 1000000;
        } else {
            s.insert(b);
            flag = a;
        }
    }
    cout << ans << endl;
    return 0;
}
