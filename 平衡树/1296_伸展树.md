```c++
#include <iostream>
#include <algorithm>

using namespace std;

const int INF = 0x7fffffff;
const int N = 100001;
int pre[N], key[N], ch[N][2];
int tot = 0;
int root = 0;


void new_node(int &r, int father, int k) {
    r = ++tot;
    pre[r] = father;
    key[r] = k;
    ch[r][0] = ch[r][1] = 0;
}

void rotate(int x) {
    int x_pre_pre = pre[pre[x]];    //x的祖父
    int x_pre = pre[x];             //x的父亲
    int x_pre_kind = ch[x_pre_pre][1] == x_pre; //x父亲是左儿子还是右儿子
    int x_kind = ch[x_pre][1] == x; //x是左儿子还是右儿子
    int x_ch_kind = !x_kind;        //x需要变化的儿子是左儿子还是右儿子
    int x_ch = ch[x][x_ch_kind];    //x需要变化的儿子

    // 确保x是非根节点
    if (x_pre) {

        // x祖父
        if (x_pre_pre) {
            ch[x_pre_pre][x_pre_kind] = x;
        }

        // x的父亲
        pre[x_pre] = x;
        ch[x_pre][x_kind] = x_ch;

        // x自己
        pre[x] = x_pre_pre;
        ch[x][x_ch_kind] = x_pre;

        // x需要变化的儿子
        if (x_ch) {
            pre[x_ch] = x_pre;
        }
    }
}

// 伸展直到x的父亲为goal
void splay(int x, int goal) {
    int x_kind;
    int x_pre_kind;
    while (pre[x] != goal) {
        // zig & zag型，对x做单旋转
        if (pre[pre[x]] == goal) {
            rotate(x);
        }
        else {
            x_kind = ch[pre[x]][1] == x;
            x_pre_kind = ch[pre[pre[x]]][1] == pre[x];
            // zig-zig & zag-zag型，对x做zig-zig&zag-zag旋转
            if (x_kind == x_pre_kind) {
                rotate(pre[x]);
                rotate(x);
            }
            // zig-zag & zag-zig型，对x做双旋转
            else {
                rotate(x);
                rotate(x);
            }
        }
    }

    // 若将x伸展到根节点，更新root
    if (goal == 0) {
        root = x;
    }
}

int insert(int k) {
    int r = root;
    int kind;

    // 如果没有root,新建树
    if (r == 0) {
        new_node(root, 0, k);
        return 1;
    }

    // 查找节点k
    while (true) {
        if (key[r] == k) {
            splay(r, 0);
            return 0;
        }
        kind = key[r] < k;
        if (ch[r][kind]) {
            r = ch[r][kind];
        } else {
            break;
        }
    }

    // 节点k不存在，则新建该节点，并对节点k做伸展
    new_node(ch[r][kind], r, k);
    splay(ch[r][kind], 0);
    return 1;
}

// 查找x左子树的最右节点，返回差值
int get_pre(int x) {
    int tmp = ch[x][0];
    if (tmp == 0) {
        return INF;
    }
    while (ch[tmp][1]) {
        tmp = ch[tmp][1];
    }
    return key[x] - key[tmp];
}

// 查找x右子树的最左节点，返回差值
int get_next(int x) {
    int tmp = ch[x][1];
    if (tmp == 0) {
        return INF;
    }
    while (ch[tmp][0]) {
        tmp = ch[tmp][0];
    }
    return key[tmp] - key[x];
}

int main() {
    int n;
    int num;
    int diff;
    int ans = 0;
    cin >> n;
    for (int i = 0; i < n; i++) {
        cin >> num;
        if (insert(num)) {
            diff = min(get_next(root), get_pre(root));
            if (diff == INF) {
                diff = num;
            }
            ans += diff;
        }
        cout << ans << endl;
    }
}
```
