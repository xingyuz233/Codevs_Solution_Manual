#include <iostream>
#include <cstring>
using namespace std;
const int N = 30001;
int a[N];
int main() {
    int n, m;
    cin >> n >> m;
    for (int i = 1; i <= n; i++) {
        a[i] = i;
    }
    int pos = 1;
    for (int k = n; k > 0; k--) {
        pos = (pos + m - 1) % k;
        if (pos == 0) {
            pos = k;
        }
        cout << a[pos] << " ";
        memcpy(a + pos, a + pos + 1, (k - pos) * sizeof(int));
    }
}
