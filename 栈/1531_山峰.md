  #include <iostream>
  #include <stack>
  
  using namespace std;
  int main()
  {
      stack<int> s;
      int n;
      int h;
      int sum = 0;
      cin >> n;
      for (int i = 0; i < n - 1; i++) {
          cin >> h;
          while (!s.empty() && s.top() < h) {
              s.pop();
          }
          s.push(h);
          sum += s.size();
      }
      cin >> h;
      cout << sum;
  
  }
